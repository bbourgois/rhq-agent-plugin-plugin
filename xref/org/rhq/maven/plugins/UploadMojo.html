56  /**
57   * Upload a freshly built RHQ Agent Plugin to an RHQ container.
58   *
59   * @author Thomas Segismont
60   */
61  @Mojo(name = "upload", defaultPhase = LifecyclePhase.PACKAGE, threadSafe = true)
62  public class UploadMojo extends AbstractMojo {
63  
64      private static final String REST_CONTENT_URI = "/rest/content";
65      private static final String UPLOAD_URI = REST_CONTENT_URI + "/fresh";
66  
67      private static final int SOCKET_CONNECTION_TIMEOUT = 1000 * 30;
68      private static final int SOCKET_READ_TIMEOUT = 1000 * 30;
69  
70      /**
71       * The build directory (root of build works).
72       */
73      @Parameter(defaultValue = "${project.build.directory}", required = true, readonly = true)
74      private File buildDirectory;
75  
76      /**
77       * The name of the generated RHQ agent plugin archive.
78       */
79      @Parameter(defaultValue = "${project.build.finalName}", required = true, readonly = true)
80      private String finalName;
81  
82      /**
83       * The scheme to use to communicate with the remote RHQ server.
84       */
85      @Parameter(defaultValue = "http")
86      private String scheme;
87  
88      /**
89       * Remote RHQ server host.
90       */
91      @Parameter(required = true)
92      private String host;
93  
94      /**
95       * Remote RHQ server port.
96       */
97      @Parameter(defaultValue = "7080")
98      private int port;
99  
100     /**
101      * Authentication user name. The user must have appropriate permissions (MANAGE_SETTINGS).
102      */
103     @Parameter(required = true)
104     private String username;
105 
106     /**
107      * Authentication password.
108      */
109     @Parameter(required = true)
110     private String password;
111 
112     /**
113      * Whether a plugin scan should be triggered on the server after upload.
114      */
115     @Parameter(defaultValue = "true")
116     private boolean startScan;
117 
118     /**
119      * Whether to fail the build if an error occurs while uploading.
120      */
121     @Parameter(defaultValue = "true")
122     private boolean failOnError;
123 
124     /**
125      * Whether to skip the execution of this mojo.
126      */
127     @Parameter(defaultValue = "false")
128     private boolean skipUpload;
129 
130     @Override
131     public void execute() throws MojoExecutionException, MojoFailureException {
132         if (skipUpload) {
133             getLog().info("Skipped execution");
134         }
135         File agentPluginArchive = getAgentPluginArchiveFile(buildDirectory, finalName);
136         if (!agentPluginArchive.exists() && agentPluginArchive.isFile()) {
137             throw new MojoExecutionException("Agent plugin archive does not exist: " + agentPluginArchive);
138         }
139 
140         // Prepare HttpClient for plugin upload
141         ClientConnectionManager httpConnectionManager = new BasicClientConnectionManager();
142         DefaultHttpClient httpClient = new DefaultHttpClient(httpConnectionManager);
143         HttpParams httpParams = httpClient.getParams();
144         HttpConnectionParams.setConnectionTimeout(httpParams, SOCKET_CONNECTION_TIMEOUT);
145         HttpConnectionParams.setSoTimeout(httpParams, SOCKET_READ_TIMEOUT);
146         httpClient.getCredentialsProvider().setCredentials(new AuthScope(host, port),
147                 new UsernamePasswordCredentials(username, password));
148 
149         HttpPost uploadContentRequest = null;
150         HttpPut moveContentToPluginsDirRequest = null;
151         try {
152             // Upload plugin content
153             URI uploadContentUri = buildUploadContentUri();
154             uploadContentRequest = new HttpPost(uploadContentUri);
155             uploadContentRequest.setEntity(new FileEntity(agentPluginArchive, ContentType.APPLICATION_OCTET_STREAM));
156             uploadContentRequest.setHeader(HttpHeaders.ACCEPT, ContentType.APPLICATION_JSON.getMimeType());
157             HttpResponse uploadContentResponse = httpClient.execute(uploadContentRequest);
158             if (uploadContentResponse.getStatusLine().getStatusCode() != HttpStatus.SC_CREATED) {
159                 handleProblem(uploadContentResponse.getStatusLine().toString());
160             }
161             JSONObject uploadContentResponseJsonObject = new JSONObject(EntityUtils.toString(uploadContentResponse
162                     .getEntity()));
163             // Read the content handle value in JSON response 
164             String contentHandle = (String) uploadContentResponseJsonObject.get("value");
165             getLog().info("Uploaded " + agentPluginArchive);
166 
167             // Request uploaded to be moved to the plugins directory
168             URI moveContentToPluginsDirUri = buildMoveContentToPluginsDirUri(contentHandle);
169             moveContentToPluginsDirRequest = new HttpPut(moveContentToPluginsDirUri);
170             moveContentToPluginsDirRequest.setHeader(HttpHeaders.ACCEPT, ContentType.APPLICATION_JSON.getMimeType());
171             HttpResponse moveContentToPluginsDirResponse = httpClient.execute(moveContentToPluginsDirRequest);
172             if (moveContentToPluginsDirResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
173                 handleProblem(moveContentToPluginsDirResponse.getStatusLine().toString());
174             }
175             if (startScan) {
176                 getLog().info("Moved uploaded content to plugins directory and triggered plugin scan");
177             } else {
178                 getLog().info("Moved uploaded content to plugins directory");
179             }
180         } catch (IOException e) {
181             handleException(e);
182         } catch (JSONException e) {
183             handleException(e);
184         } catch (URISyntaxException e) {
185             handleException(e);
186         } finally {
187             if (uploadContentRequest != null) {
188                 uploadContentRequest.abort();
189             }
190             if (moveContentToPluginsDirRequest != null) {
191                 moveContentToPluginsDirRequest.abort();
192             }
193             httpConnectionManager.shutdown();
194         }
195     }
196 
197     private void handleProblem(String message) throws MojoExecutionException {
198         if (failOnError) {
199             throw new MojoExecutionException(message);
200         }
201         getLog().error(message);
202     }
203 
204     private void handleException(Exception e) throws MojoExecutionException {
205         if (failOnError) {
206             throw new MojoExecutionException(e.getMessage(), e);
207         }
208         getLog().error(e.getMessage(), e);
209     }
210 
211     private URI buildUploadContentUri() throws URISyntaxException {
212         return new URIBuilder() //
213                 .setScheme(scheme) //
214                 .setHost(host) //
215                 .setPort(port) //
216                 .setPath(UPLOAD_URI) //
217                 .build();
218     }
219 
220     private URI buildMoveContentToPluginsDirUri(String contentHandle) throws URISyntaxException {
221         return new URIBuilder() //
222                 .setScheme(scheme) //
223                 .setHost(host) //
224                 .setPort(port) //
225                 .setPath(REST_CONTENT_URI + "/" + contentHandle + "/plugins") //
226                 .setParameter("name", getAgentPluginArchiveFile(buildDirectory, finalName).getName()) //
227                 .setParameter("startScan", String.valueOf(startScan)) //
228                 .build();
229     }
