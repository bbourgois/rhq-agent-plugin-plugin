44  public abstract class AbstractExecCliMojo extends AbstractMojo {

46      private static final String RHQ_CLI_MODULE_GROUP_ID = "org.rhq";
47      private static final String RHQ_CLI_MODULE_ARTIFACT_ID = "rhq-remoting-cli";
48      private static final String RHQ_CLI_SCRIPT_WINDOWS = "rhq-cli.bat";
49      private static final String RHQ_CLI_SCRIPT_OTHERS = "rhq-cli.sh";

51      /**
52       * The build directory (root of build works).
53       */
54      @Parameter(defaultValue = "${project.build.directory}", required = true, readonly = true)
55      protected File buildDirectory;
56  
57      /**
58       * The CLI directory (where to install the RHQ CLI).
59       */
60      @Parameter(defaultValue = "${project.build.directory}/rhq-cli", required = true)
61      protected File rhqCliDirectory;
62  
63      /**
64       * Version of the RHQ CLI.
65       */
66      @Parameter(required = true)
67      protected String rhqVersion;
68  
69      /**
70       * Whether to login to a remote RHQ server.
71       */
72      @Parameter(defaultValue = "false")
73      protected boolean login;
74  
75      /**
76       * Remote RHQ server host. Required if <code>login</code> is set to true.

78      @Parameter(required = false)
79      protected String host;
80  
81      /**
82       * Remote RHQ server port.
83       */
84      @Parameter(defaultValue = "7080")
85      protected int port;
86  
87      /**
88       * Authentication user name. Required if <code>login</code> is set to true.

90      @Parameter(required = false)
91      protected String username;
92  
93      /**
94       * Authentication password. Required if <code>login</code> is set to true.
95       */
96      @Parameter(required = false)
97      protected String password;
98  
99      /**
100      * Whether to fail the build if an error occurs while uploading.
101      */
102     @Parameter(defaultValue = "true")
103     protected boolean failOnError;
104 
105     @Parameter(defaultValue = "${project.remoteArtifactRepositories}", required = true, readonly = true)
106     private List remoteRepositories;
107 
108     @Parameter(defaultValue = "${localRepository}", required = true, readonly = true)
109     private ArtifactRepository localRepository;
110 
111     @Component
112     private ArtifactFactory artifactFactory;
113 
114     @Component
115     private ArtifactResolver artifactResolver;
116 
117     @Override
118     public void execute() throws MojoExecutionException, MojoFailureException {
119         if (shouldSkip()) {
120             getLog().info("Skipped execution");
121         }
122         validateCommonParams();
123         validateParams();
124         installRhqCLi();
125         doExecute();
126     }
127 
128     protected abstract boolean shouldSkip();
129 
130     private void validateCommonParams() throws MojoExecutionException {
131         if (login) {
132             if (StringUtils.isBlank(host)) {
133                 throw new MojoExecutionException("'host' param is blank");
134             }
135             if (StringUtils.isBlank(username)) {
136                 throw new MojoExecutionException("'username' param is blank");
137             }
138             if (StringUtils.isBlank(password)) {
139                 throw new MojoExecutionException("'password' param is blank");
140             }
141         }
142     }
143 
144     protected abstract void validateParams() throws MojoExecutionException, MojoFailureException;
145 
146     private void installRhqCLi() throws MojoExecutionException {
147         Artifact rhqCliZipArtifact = this.artifactFactory.createArtifact(
148                 RHQ_CLI_MODULE_GROUP_ID, RHQ_CLI_MODULE_ARTIFACT_ID, rhqVersion,
149                 null, "zip");
150         try {
151             artifactResolver.resolve(rhqCliZipArtifact, remoteRepositories, localRepository);
152             ZipUtil.unpack(rhqCliZipArtifact.getFile(), rhqCliDirectory);
153             if (getLog().isDebugEnabled()) {
154                 getLog().debug("Installed RHQ CLI");
155             }
156         } catch (Exception e) {
157             throw new MojoExecutionException(e.getMessage(), e);
158         }
159     }
160 
161     protected abstract void doExecute() throws MojoExecutionException, MojoFailureException;
162 
163     protected File getRhqCliStartScriptFile() throws IOException {
164         String scriptName;
165         if (System.getProperty("os.name").toLowerCase().indexOf("win") != -1) {
166             scriptName = RHQ_CLI_SCRIPT_WINDOWS;
167         } else {
168             scriptName = RHQ_CLI_SCRIPT_OTHERS;
169         }
170         List files = FileUtils.getFileNames(rhqCliDirectory, "**/" + scriptName, null, false, true);
171         if (files.size() == 1) {
172             String filename = (String) files.get(0);
173             if (getLog().isDebugEnabled()) {
174                 getLog().debug("Found RHQ CLI start script: " + filename);
175             }
176             File file = new File(rhqCliDirectory, filename);
177             if (!file.canExecute()) {
178                 file.setExecutable(true);
179             }
180             return file;
181         } else if (files.size() > 1) {
182             getLog().warn("Found more than one RHQ CLI start Script: " + files);
183         }
184         return null;
185     }
186 
187     protected void handleFailure(String message) throws MojoFailureException {
188         if (failOnError) {
189             throw new MojoFailureException(message);
190         }
191         getLog().error(message);
192     }
193 
194     protected void handleException(Exception e) throws MojoExecutionException {
195         if (failOnError) {
196             throw new MojoExecutionException(e.getMessage(), e);
197         }
198         getLog().error(e.getMessage(), e);
199     }
