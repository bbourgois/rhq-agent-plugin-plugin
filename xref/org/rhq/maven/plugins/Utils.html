44      public static File getAgentPluginArchiveFile(File buildDirectory, String finalName) {
45          return new File(buildDirectory, finalName + ".jar");
46      }
47  
48      public static Set<File> findParentPlugins(MavenProject agentPluginProject) throws IOException {
49          Set<File> parentPlugins = new HashSet<File>();
50          Iterator projectArtifacts = agentPluginProject.getArtifacts().iterator();
51          ScopeArtifactFilter artifactFilter = new ScopeArtifactFilter(Artifact.SCOPE_COMPILE);
52          while (projectArtifacts.hasNext()) {
53              Artifact artifact = (Artifact) projectArtifacts.next();
54              if (!artifact.isOptional() && artifact.getType().equals("jar") && artifactFilter.include(artifact)) {
55                  File artifactFile = artifact.getFile();
56                  if (isAgentPlugin(artifactFile)) {
57                      parentPlugins.add(artifactFile);
58                  }
59              }
60          }
61          return parentPlugins;
62      }
63  
64      public static boolean isAgentPlugin(File jarFile) {
65          return ZipUtil.containsEntry(jarFile, "META-INF/rhq-plugin.xml");
66      }
67  
68      public static void redirectOuput(Process process) {
69          startCopyThread(process.getInputStream(), System.out);
70          startCopyThread(process.getErrorStream(), System.err);
71      }
72  
73      public static void startCopyThread(final InputStream inputStream, final PrintStream printStream) {
74          Thread thread = new Thread(new Runnable() {
75              @Override
76              public void run() {
77                  try {
78                      copy(inputStream, printStream);
79                  } catch (IOException ignore) {
80                  }
81              }
82          });
83          thread.setDaemon(true);
84          thread.start();
85      }
